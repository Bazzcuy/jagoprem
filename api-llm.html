chat_example.py 
from openai import OpenAI

client = OpenAI(
    api_key="sk-your-api-key",
    base_url="https://api.jagoprem.shop/v1"
)

# Chat Completion
response = client.chat.completions.create(
    model="openai/gpt-4o",  # atau "anthropic/claude-sonnet-4-5"
    messages=[
        {"role": "system", "content": "Kamu adalah asisten yang membantu."},
        {"role": "user", "content": "Jelaskan apa itu LLM dalam 2 kalimat."}
    ],
    max_tokens=512,
    temperature=0.7
)

print(response.choices[0].message.content)

# Embeddings
embed = client.embeddings.create(
    model="openai/text-embedding-3-small",
    input="Teks untuk diembedding"
)
print(embed.data[0].embedding[:5])  # Print 5 dimensi pertama

chat.js 
import OpenAI from 'openai';

const client = new OpenAI({
  apiKey: 'sk-your-api-key',
  baseURL: 'https://api.jagoprem.shop/v1',
});

async function main() {
  // Non-streaming
  const res = await client.chat.completions.create({
    model: 'anthropic/claude-sonnet-4-5',
    messages: [{ role: 'user', content: 'Hi!' }],
  });
  console.log(res.choices[0].message.content);

  // Streaming
  const stream = await client.chat.completions.create({
    model: 'google/gemini-2.5-flash',
    messages: [{ role: 'user', content: 'Ceritakan sebuah kisah pendek!' }],
    stream: true,
  });
  for await (const chunk of stream) {
    process.stdout.write(chunk.choices[0]?.delta?.content ?? '');
  }
}

main();
chat.php 
<?php

$apiKey = 'sk-your-api-key';
$baseUrl = 'https://api.jagoprem.shop/v1';

$payload = json_encode([
    'model' => 'openai/gpt-4o-mini',
    'messages' => [[
        'role' => 'user',
        'content' => 'Halo dari PHP!'
    ]]
]);

$ch = curl_init($baseUrl . '/v1/chat/completions');
curl_setopt($ch, CURLOPT_HTTPHEADER, [
    'Authorization: Bearer ' . $apiKey,
    'Content-Type: application/json'
]);
curl_setopt($ch, CURLOPT_POSTFIELDS, $payload);
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);

$response = curl_exec($ch);
$data = json_decode($response, true);

echo $data['choices'][0]['message']['content'];
